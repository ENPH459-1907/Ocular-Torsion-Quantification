[docs]def signal_offset(sig1, sig2):
    if len(sig1) != len(sig2):
        raise DifferentSignalShapeError('Sig1 and Sig2 must be of the same length.')

    l = len(sig1)
    c = signal.correlate(sig1, sig2)
    offset = l - np.argmax(c)

    return offset
